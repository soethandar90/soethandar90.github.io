Array.prototype.bSort = function () {
const a = this.slice();
const n = a.length;
for (let i = 0; i < n - 1; i++) {
for (let j = 0; j < n - i - 1; j++) {
if (a[j] > a[j + 1]) {
const temp = a[j];
a[j] = a[j + 1];
a[j + 1] = temp;
}
}
}
return a;
};
