String.prototype.filter = function (bwords) {
if (!Array.isArray(bwords) || bwords.length === 0) {
return this.toString();
}
let filString = this.toString();
bwords.forEach((word) => {
const bWordExp = new RegExp("\\b" + word + "\\b", "gi");
filString = filString.replace(bWordExp, '');
});
filString = filString.replace(/\s+/g, ' ').trim();
return filString;
};
